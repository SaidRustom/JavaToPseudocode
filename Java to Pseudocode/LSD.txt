/*
 * class PotatoChipBag starts
 */
public class PotatoChipBag {
	private int size; // Regular or Large (use constants below)
	private double weight; // oz
	public static final int REGULAR = 1;
	private static final double REGULAR_WEIGHT = 9.25; // oz
	public static final int LARGE = 2;
	private static final double LARGE_WEIGHT = 15.75; // oz
	private static final double EPSILON = 0.01;
	
	
	 /*  no-argument constructor sets fields to REGULAR, REGULAR_WEIGHT
	 */
	public PotatoChipBag() {
		this(REGULAR, REGULAR_WEIGHT);
	}
	
	/*
	 * overloaded constructor accepting values for fields
	 */
	public PotatoChipBag(int size, double weight) {
		this.size = size;
		this.weight = weight;
	}
	
	/*
	 * gettor for size
	 */
	public int getSize() {
		return size;
	}
	/*
	 * settor for size
	 */
	public void setSize(int size) {
		this.size = size;
	}
	
	/*
	 * gettor for weight
	 */
	public double getWeight() {
		return weight;
	}
	
	/*
	 * settor for weight
	 */
	public void setWeight(double weight) {
		this.weight = weight;
	}

	/*
	 * method isBagCorrectWeight verifies if the bag is within tolerance of 0.1
	 * based on size of bag and weight
	 */
	public boolean isBagCorrectWeight() {
		boolean result = true; // can be changed to true here if needed
		double difference;
		if (getSize() == REGULAR) {
			difference = Math.abs(REGULAR_WEIGHT - getWeight());
			if (difference <= EPSILON) {
				result = true;	
			}
			else {result = false;
			}
		}else if (getSize() == LARGE) {
			difference = Math.abs(LARGE_WEIGHT - getWeight());
			if (difference <= EPSILON) {
				result = true;
			}
			else { result = false;
			}
		}
		
		return result;
	}
	
}
